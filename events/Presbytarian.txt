namespace = presbytarian_spawn_event
normal_or_historical_nations = yes

# The Reformation Branches Out
country_event = {
	id = presbytarian_spawn_event.1
	title = "presbytarian_spawn_event.1.t"
	desc = "presbytarian_spawn_event.1.d"
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = LUTHER_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = RELIGION_eventPicture
	}
	
	major = yes
	
	trigger = {
		is_religion_enabled = reformed
		religion_years = { reformed = 15 }
		NOT = { is_religion_enabled = presbytarian }
		is_defender_of_faith = no
		OR = {
			religion = catholic
			religion = protestant
			religion = reformed
		}
		NOT = { has_reform = papacy_reform }
		OR = {
			primary_culture = scottish
			primary_culture = northumbrian
		}
	}
	
	mean_time_to_happen = {
		months = 80
		
		modifier = {
			factor = 0.5
			religion_years = { reformed = 15 }
		}
		modifier = {
			factor = 0.5
			religion_years = { reformed = 30 }
		}
	}
	
	option = {		#Spawn of the Presbytarians
		name = "presbytarian_spawn_event.1.a"
		enable_religion = presbytarian 
		random_owned_province = {
			limit = {
				OR = {
					culture = scottish
					culture = northumbrian
				}
			}
			change_religion = presbytarian
		}
		ROOT = {
			capital_scope = {
				change_religion = presbytarian
				add_permanent_province_modifier = {
					name = "religious_zeal_at_conv"
					duration = 9000
				}
			}
		}
		set_country_flag = "presbytarian_reformation" 
		hidden_effect = {
			if = {
				limit = { NOT = { any_owned_province = { religion = presbytarian is_reformation_center = yes } } }
				capital_scope = {
					if = {
						limit = { NOT = { religion = presbytarian } }
						change_religion = presbytarian
					}
					add_reform_center = presbytarian
				}
			}
		}
	}
}
