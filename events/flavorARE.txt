country_event = {			# Waldensians of Lyon
	id = flavor_are.1
	title = flavor_are.1.t
	desc = flavor_are.1.d
	picture = PRAYING_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		tag = ARE
	}	
	option = {		# Do not renew the treaty of autonomy
		name = flavor_are.1.a
		set_country_flag = waldensians_not_allowed_autonomy
		remove_province_modifier = waldensian_autonomy_mod
		custom_tooltip = waldensians_not_allowed_continued_autonomy.tt
		every_owned_province = {
			limit = {
				religion = waldensian
			}
			spawn_rebels = {
				type = waldensian_rebels
				size = 1
			}
		}
	}
	option = {		# Allow them continued autonomy!
		name = flavor_are.1.b
		trigger = { ai = no }
		set_country_flag = waldensians_allowed_autonomy
		#custom_tooltip = waldensians_allowed_continued_autonomy.tt
		every_owned_province = {
			limit = {
				religion = waldensian
			}
			add_province_modifier = {
				name = waldensian_autonomy_mod
				duration = -1				
			}
		}
	}
}

country_event = {			#Unification of the Burgundies
	id = flavor_are.2
	title = flavor_are.2.t
	desc = flavor_are.2.d
	picture = REFORM_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		tag = ARE
	}	
	
	option = {
		name = flavor_are.2.a 
		
		change_primary_culture = arleat
		custom_tooltip = are_become_arleat.tt
		hidden_effect = {
			every_province = {
				limit = {
					OR = {
						culture = burgundian
						culture = arpitan
						culture = provencal
					}	
				}
				change_culture = arleat
			}
			set_ruler_culture = arleat
			if = {
				limit = {
					has_heir = yes
				}
				set_heir_culture = arleat
			}
			if = {
				limit = {
					has_consort = yes
					OR = {
						consort_culture = burgundian
						consort_culture = arpitan
						consort_culture = provencal
					}
				}
				set_consort_culture = arleat
			}
			every_country = {
				limit = {
					OR = {
						culture = burgundian
						culture = arpitan
						culture = provencal
					}
				}
				change_primary_culture = arleat
				set_ruler_culture = arleat
				if = {
					limit = {
						has_heir = yes
					}
					set_heir_culture = arleat
				}
				if = {
					limit = {
						has_consort = yes
						OR = {
							consort_culture = burgundian
							consort_culture = arpitan
							consort_culture = provencal
						}
					}
					set_consort_culture = arleat
				}
			}
		}
	}
}
